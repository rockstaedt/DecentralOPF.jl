%%% Sets %%%
\nomenclature[S]{\(\mathcal{G}\)}{Set of generators}
\nomenclature[S]{\(\mathcal{S}\)}{Set of storages}
\nomenclature[S]{\(\mathcal{T}\)}{Set of hourly timesteps}
\nomenclature[S]{\(\mathcal{N}\)}{Set of nodes}
\nomenclature[S]{\(\mathcal{L}\)}{Set of transmission lines}

%%% Variables %%%
\nomenclature[V]{$P_{g,t}$}{Power generation of generator $g$ at timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[V]{$D_{s,t}$}{Withdrawal power of storage $s$ at timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[V]{$C_{s,t}$}{Charging power of storage $s$ at timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[V]{$E_{s,t}$}{Storage level of storage $s$ at timestep $t$\nomunit{\SI{1}{\mega\watt\hour}}}
\nomenclature[V]{$\lambda_t$}{Dual variable of the system balance constraint}
\nomenclature[V]{$\hat{\lambda}_t$}{Scaled dual variable of the system balance constraint}
\nomenclature[V]{$\mu_{l,t}$}{Dual variable of the upper power flow constraint}
\nomenclature[V]{$\hat{\mu}_{l,t}$}{Scaled dual variable of the upper power flow constraint}
\nomenclature[V]{$\rho_{l,t}$}{Dual variable of the lower power flow constraint}
\nomenclature[V]{$\hat{\rho}_{l,t}$}{Scaled dual variable of the lower power flow constraint}
\nomenclature[V]{$U_{l,t}$}{Slack variable for the upper capacity of transmission line $l$\\ at timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[V]{$K_{l,t}$}{Slack variable for the lower capacity of transmission line $l$\\ at timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[V]{$r_{n,t}$}{Nodal price of node $n$\\ at timestep $t$\nomunit{\SI{1}{EUR}}}

%%% Parameters %%%
\nomenclature[P]{$d_{n,t}$}{Demand at node $n$ and timestep $t$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\overline{p_g}$}{Maximum power of generator $g$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\overline{p_s}$}{Maximum withdrawal and charging power of storage $s$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\overline{e_s}$}{Maximum energy level of storage $s$\nomunit{\SI{1}{\mega\watt\hour}}}
\nomenclature[P]{$\overline{f_l}$}{Maximum line capacity of transmission line $l$\nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$mc_g$}{Marginal costs of generator $g$ \nomunit{\SI{1}{EUR/\mega\watt}}}
\nomenclature[P]{$mc_s$}{Marginal costs of storage $s$ \nomunit{\SI{1}{EUR/\mega\watt}}}
\nomenclature[P]{$\Theta_{n,t}$}{Sum of the previous iteration power values' of all \\ generators located at node $n$ and timestep $t$ \nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\Phi_{n,t}$}{Sum of the previous iteration discharge power values' of all\\ storages located at node $n$ and timestep $t$ \nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\Psi_{n,t}$}{Sum of the previous iteration charge values' of all\\ storages located at node $n$ and timestep $t$ \nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\Upsilon_{l,t}$}{Average of slack variable $U_{l,t}$ for the upper limit of the previous iteration \nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\Gamma_{l,t}$}{Average of slack variable $K_{l,t}$ for the lower limit of the previous iteration \nomunit{\SI{1}{\mega\watt}}}
\nomenclature[P]{$\epsilon$}{Threshold term for the convergence of the algorithm \nomunit{$10^{-3}$}}
